# promise 其他静态方法

# polyfill Promise.first()

if (!Promise.first) {
  Promise.first = function(prs) {
    return new Promise(function(resolve, reject) {
      prs.forEach(function(pr) {
        Promise.resolve(pr).then(resolve);
      });
    });
  };
}

# 串行期约合成

多个函数合成一般这么做：

function addTwo(x) {
  return x + 2;
}
function addThree(x) {
  return x + 3;
}
function addFive(x) {
  return x + 5;
}

function addTen(x) {
  return addFive(addTwo(addThree(x)));
}
console.log(addTen(7)); // 17

# 通过 reduce 改进：

function addTen(x) {
  return [addTwo, addThree, addFive].reduce(
    (promise, fun) => promise.then(fun),
    Promise.resolve(x)
  );
}
addTen(8).then(console.log); // 18

# 通过async和await改进：

async function addTen(x) {
    for(fn of [addTwo, addThree, addFive]) {
        x = await fn(x);

# 期约取消的实现

<button id="start">
  Start
</button>
<button id="cancel">
  Cancel
</button>

<script>
  class CancelToken {
    constructor(callback) {
      this.promise = new Promise((resolve, reject) => {
        callback(() => {
          resolve(); // 在回调中传入函数，使得期约强制进入resolve
        });
      });
    }
  }

  const startButton = document.querySelector("#start");
  const cancelButton = document.querySelector("#cancel");

  function cancelClick(delay) {
    return new Promise((resolve, reject) => {
      const id = setTimeout(() => {
        console.log("延迟发生点击");
      }, delay);

      const cancelClick = new CancelToken((callback1) => {
        // callback1为class CancelToken中传入的回调
        cancelButton.addEventListener("click", callback1);
      });
      cancelClick.promise.then(() => clearTimeout(id));
    });
  }

  startButton.addEventListener("click", () => cancelClick(1000));
</script>

# 期约进度通知计时案例

class TrackablePromise extends Promise {
  constructor(executor) {
    // 打印通知的函数
    // 把函数暂存在数组中，之后通过map循环调用
    const notifyHandlers = [];
    super((resolve, reject) => {
      return executor(resolve, reject, (status) => {
        // TrackablePromise的第三个参数为回调函数
        // 回调函数的第一个参数将会传入notifyHandlers中的暂存函数并调用
        notifyHandlers.map((handler) => handler(status));
      });
    });
    this.notifyHandlers = notifyHandlers;
  }
  notify(notifyHandler) {
    this.notifyHandlers.push(notifyHandler);
    return this;
  }
}

let p = new TrackablePromise((resolve, reject, notify) => {
  // 执行异步操作
  function countdown(x) {
    if (x > 0) {
      // 将${20 * x}% remaining作为参数
      // 传入notifyHandlers中的函数中调用
      notify(`${20 * x}% remaining`);
      // 异步操作
      setTimeout(() => countdown(x - 1), 1000);
    } else {
      resolve();
    }
  }
  countdown(5);
});

// 将通知函数push
p.notify((x) => setTimeout(console.log, 0, "progress:", x));
p.then(() => setTimeout(console.log, 0, "completed"));
