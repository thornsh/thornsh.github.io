# v-cloak

这个指令可以隐藏未编译的 Mustache 标签直到组件实例准备完毕

<body>
  <div id="app"></div>

  <template id="my-app">
    <h2 v-cloak>{{message}}</h2>
  </template>

  <script src="../js/vue.js"></script>
  <script>
    const App = {
      template: "#my-app",
      data() {
        return {
          message: "Hello World",
        };
      },
    };

    Vue.createApp(App).mount("#app");
  </script>
</body>
