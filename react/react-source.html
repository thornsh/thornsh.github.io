# pureComponent 的继承

function ComponentDummy() {}
// 使pureComponent与Component公用一套方法
ComponentDummy.prototype = Component.prototype;
function PureComponent(props, context, updater) {
    ...;
}
// 使PureComponent.prototype为一个空对象（因为ComponentDummy()为空）,
// 并将PureComponent.prototype的[[proto]]指向Component.prototype
// const pureComponentPrototype =
// (PureComponent.prototype = Object.create(ComponentDummy.prototype));
const pureComponentPrototype = (PureComponent.prototype = new ComponentDummy());
// 修改constructor的值
pureComponetnPrototype.constructor = PureComponent;

// 浅拷贝，将Component.prototype里的方法属性拷贝到PureComponent.prototype里面
// 之前PureComponent.prototype.__proto__里才有Component.prototype里的方法
// 这样做减少了原型查找的额外时间
// 之后，PureComponent.prototype里有方法，且Component.prototype里的方法改变后不会
// 改变PureComponent.prototype浅拷贝的方法，但PureComponent.prototype.__proto__
// 会改变
Object.assign(pureComponentPrototype, Component.prototype);

pureComponentPrototype.isPureReactComponent = true;
