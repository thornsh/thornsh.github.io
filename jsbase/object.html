# 对象方法

# 禁止扩展 Object.preventExtensions()

如果想禁止一个对象添加新属性并且保留已有属性，可以使用 Object.preventExtensions()

var myObject = {
  a: 2,
};
Object.preventExtensions(myObject);
myObject.b = 3;
myObject.b; // undefined

# 密封 Object.seal()

这个方法本质上调用 Object.preventExtensions()并把所有现有的属性标记为configurable: false

所以密封之后不能添加新属性，并且不能重新配置或者删除现有属性,但是可以修改属性的值

# 冻结 Object.freeze()

这个方法在密封的基础上把所有“数据访问”属性标记为writable: false，使得无法修改值

# Object.isExtensible()

对象是否可扩展

// 新对象默认是可扩展的.
var empty = {};
Object.isExtensible(empty); // === true

// ...可以变的不可扩展.
Object.preventExtensions(empty);
Object.isExtensible(empty); // === false

// 密封对象是不可扩展的.
var sealed = Object.seal({});
Object.isExtensible(sealed); // === false

// 冻结对象也是不可扩展.
var frozen = Object.freeze({});
Object.isExtensible(frozen); // === false

# in 操作符

in 操作符会检查属性是否存在对象以及其[[Prototype]]原型链中。

# hasOwnProperty()

hasOwnProperty()则只会检查属性是否存在当前对象中，并不会检查原型链

# Object.keys()

只会查找当前对象，返回可枚举的属性数组

# Object.getOwnPropertyNames()

只会查找当前对象，返回所有属性数组，不管是否可枚举

TIP

因为 hasOwnProperty()是存在对象 prototype 上的方法，有些对象有可能没有连接到 Object.prototype,(比如通过 Object.create(null)来创建的对象)，这时候可以使用：Object.prototype.hasOwnProperty.call(myObject, "a")来判断是否有属性 a

WARNING

in 操作符判断的是属性名，4 in [1,2,3,4] 并不是 TRUE

# Object.assign()

将源对象中可枚举和自有的属性复制到目标对象上

会使用源对象上的getter取得属性的值，然后使用目标对象上的setter设置属性的值

dest = {
    set a(value) {
        console.log(`xxx${value}`);
    }
};
src = {
    get a() {
        console.log("getter");
        return "foo";
    }
};
Object.assign(dest, src);

# Object.is()相等判断

console.log(Object.is(true, 1)); // false
console.log(Object.is({}, {})); // false
console.log(Object.is("2", 2)); // false

console.log(Object.is(+0, -0)); // false
console.log(Object.is(+0, 0)); // true
console.log(Object.is(-0, 0)); // false 

console.log(Object.is(NaN, NaN)); // true

# 工厂模式

虽然解决了创建多个类似对象的问题，但是没有解决对象标识问题（即新创建的对象是什么类型）

function createPerson(name, age, job) {
    let o = new Object();
    o.name = name;
    o.age = age;
    o.job = job;
    o.sayName = function() {
        console.log(this.name);
    };
    return o;
}

let person1 = createPerson("aaa", 18, "aJob");
let person2 = createPerson("bbb", 20, "bJob");

# 构造函数模式

问题在于其定义的方法会在每个实例上都创建一遍

通过new操作符调用构造函数，会进行如下操作

  1. 在内存中创建一个新对象
  2. 这个新对象的[[Prototype]]指向构造函数的prototype属性
  3. 构造函数中的this被赋值为这个新对象
  4. 执行构造函数内部的代码，给新对象添加属性
  5. 如果构造函数返回非空对象，则返回该对象，否则返回刚创建的新对象
function Person(name, age, job) {
    this.name = name;
    this.age = age;
    this.job = job;
    this.sayName = function() {
        console.log(this.name);
    };
}

let person1 = new Person("aaa", 18, "aJob");

# 原型模式

每个函数都会创建一个prototype属性，这个属性是一个对象，可以共享属性和方法

只要创建一个函数，就会为这个函数创建一个prototype属性，并且该原型对象会自动获得一个constructor属性，指回与之关联的构造函数

# 继承方案

# 组合继承

结合了原型链和盗用构造函数，使用原型链继承原型上的属性和方法，通过盗用构造函数继承实例属性

问题在于父类构造函数始终会被调用两次

function SuperType(name) {
    this.name = name;
    this.colors = ["red", "blue", "green"];
}
SuperType.prototype.sayName = function() {
    console.log(this.name);
}
function SubType(name, age) {
    // 继承属性
    SuperType.call(this, name);
    this.age = age;
}
// 继承方法
SubType.prototype = new SuperType();
SubType.prototype.sayAge = function() {
    console.log(this.age);
};

# 寄生式组合继承

function SuperType(name) {
    this.name = name;
    this.colors = ["red", "blue", "green"];
}
SuperType.prototype.sayName = function() {
    console.log(this.name);
};

function SubType(name, age) {
    SuperType.call(this, name);
    this.age = age;
}
SubType.prototype = Object.create(SuperType.prototype);
Object.defineProperty(SubType.prototype, "constructor", {
    enumerable: false,
    writable: true,
    configurable: true,
    value: SubType // SubType
})
SubType.prototype.sayAge = function() {
    console.log(this.age);
};

 foo() {
 console.log('foo');
 }

let BarMixin = (Superclass) => class extends Superclass {
 bar() {
 console.log('bar');
 }
};
let BazMixin = (Superclass) => class extends Superclass {
 baz() {
 console.log('baz');
 }
};
function mix(BaseClass, ...Mixins) {
 return Mixins.reduce((accumulator, current) => current(accumulator), BaseClass);
}
class Bus extends mix(Vehicle, FooMixin, BarMixin, BazMixin) {}
let b = new Bus();
b.foo(); // foo
b.bar(); // bar
b.baz(); // baz
