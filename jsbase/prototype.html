# [[Prototype]]

# 属性设置和屏蔽

myObject.foo = "bar";
1

如果 foo 不是直接存在于 myObject 中，[[Prototype]]链就会被遍历。

  1. 如果原型链上找不到 foo，foo 就会被直接添加到 myObject 上。
  2. 如果 foo 存在于原型链上层
    • 如果[[Prototype]]链上层存在 foo 属性，并且 writable：true。那么会直接在 myObject 中添加一个 foo 属性，他是屏蔽属性
    • 如果 writable：false。那么无法创建
    • 如果[[Prototype]]链上层存在 foo 且它是 setter，那么就一定会调用这个 setter，且 foo 不会被添加到 myObject 中，也不会重新定义 foo 这个 setter

TIP

如果希望最后两点也屏蔽 foo，那么就不能使用=来赋值，而是使用Object.defineProperty()

# 注意

var anotherObject = {
    a: 2
}
var myObject = Object.create(anotherObject);
anotherObject.a; // 2
myObject.a; // 2

anotherObject.hasOwnProperty("a"); // true
myObject.hasOwnProperty("a"); // false

myObject.a++; // 隐式屏蔽 相当于 myObject.a = myObject.a + 1 myObject自己创建了a

anotherObject.a; // 2
myObject.a; // 3

myObject.hasOwnProperty("a"); // true

# 原型继承（寄生式组合继承）













 



















function Foo(name) {
    this.name = name;
}
Foo.prototype.myName = function() {
    return this.name;
}
function Bar(name, label) {
    Foo.call(this, name);
    this.label = label;
}

// 创建Bar.prototype对象关联到Foo.prototype
Bar.prototype = Object.create(Foo.prototype);

// 因为Bar.prototype指向了新对象，失去了constructor，需要手动设置
Object.defineProperty(Bar.prototype, "constructor", {
    enumerable: false,
    writable: true,
    configurable: true,
    value: Bar // 使constructor指向Foo
})

// 创建Bar自己的方法
Bar.prototype.myLabel = function() {
    return this.label;
}

var a = new Bar("a", "obj a");

a.myName(); // "a"
a.myLabel(); // "obj a"

# __proto__实现

Object.defineProperty(Object.prototype, "__proto__", {
    set(newValue) {
        Object.setPrototypeOf(this, newValue);
    },
    get() {
        return Object.getPrototypeOf(this);
    }
})

# 类理论

class Task {
    constructor(id) {
        this.id = id;
    }
    outputTask() {
        console.log(this.id);
    }
}
class AAA extends Task {
    constructor(id, label){
        super(id);
        this.label = label;
    }
    outputTask() {
        super().outputTask();
        console.log(this.label);
    }
}

# 委托理论

Task = {
    setId(id) {
        this.id = id;
    },
    outputId() {
        console.log(this.id);
    }
}
AAA = Object.create(Task);

AAA.init = function(id, label) {
    this.setId(id);
    this.label = label;
}
AAA.output = function() {
    this.outputId();
    console.log(this.label);
}

# 检查对象之间的关联

function Foo() {};
Foo.prototype = {};

function Bar() {};
Bar.prototype = Object.create(Foo.prototype);

var b1 = new Bar("b1");

Bar.prototype instanceof Foo; // true
Object.getPrototypeOf(Bar.prototype) === Foo.prototype; //true
Foo.prototype.isPrototypeOf(Bar.prototype); // true

b1 instanceof Foo; // true
b1 instanceof Bar; //true
Object.getPrototypeOf(b1) === Bar.prototype; // true
Foo.prototype.isPrototypeOf(b1); // true
Bar.prototype.isPrototypeOf(b1); // true
