# yield*

能够迭代一个可迭代对象，从而一次产生出一个值

function* generatorFn() {
    for(const x of [1,2,3]) {
        yield x;
    }
}
// 等价于
function* generatorFn() {
    yield* [1,2,3];
}

# 辅助函数

function step(gen) {
  var it = gen();
  var last;
  return function() {
    // 将yield的值返回给yield前面的赋值表达式
    last = it.next(last).value;
  };
}

# 用例异步函数

function getDataAsync(url) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            var res = {
                url,
                data: Math.random();
            }
            resolve(res);
        }, 1000)
    })
}

# 执行器的封装

function run(g) {
  var gen = g();
  function next(data) {
    var res = gen.next(data);
    if (gen.done) {
      return res.value;
    }
    res.value.then((data) => {
      gen.next(data);
    });
  }
  next();
}

# 生成器版本

function* Iterator() {
  var currentValue = 0;
  while (currentValue < 4) {
    currentValue++;
    yield currentValue;
  }
}

for (var i of Iterator()) {
  console.log(i);

# 原始版本





 
 
 
















var Iterable = (function() {
  var currentValue = 0;
  return {
    // for...of循环需要
    [Symbol.iterator]: function() {
      return this;
    },
    // 标准迭代器接口
    next: function() {
      currentValue++;
      if (currentValue > 3) {
        // 不会被遍历
        return { done: true, value: undefined };
      } else {
        return { done: false, value: currentValue };
      }
    },
  };
})();
for (var i of Iterable) {
  console.log(i); // 1 2 3
}

TIP

for..of 循环在每次迭代中自动调用 next()，它不会向 next() 传入任何值，并且会在接收到 done:true 之后自动停止

WARNING

5-7 行直接返回 this 是因为：
for...of 这些迭代操作本质上是先找到对象的[Symbol.iterator]方法并创建迭代器，之后调用这个迭代器的 next()方法

# thunk 形实转换程序

一个用于调用另外一个函数的函数

# 同步 thunk

function foo(x, y) {
  return x + y;
}
function thunk() {
  return foo(3, 4);
}

console.log(thunk());

# 异步 thunk

接收一个回调

function foo(x, y, cb) {

# 封装 thunk

thunk 第一个参数为最后调用的函数，其余的参数为最后调用函数的参数

function thunkFun(fun) {
  var args = [].slice.call(arguments, 1); // 取第一个参数之后的参数
  return function(cb) {
    args.push(cb);
    return fun.apply(null, args);
  };
}
var thunk = thunkFun(foo, 3, 4);
thunk(function(sum) {
  console.log(sum); // 7
});
