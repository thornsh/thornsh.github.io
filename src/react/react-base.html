# context

  • 通过 Context.Provider 的 value 属性，将 context 的值传递给消费子组件

  • 通过 Class.contextType 赋值，使当前 class 组件能通过 this.context 访问最近的 Context.Provider 里的 value 值，如果组件上层找不到 Provider，则使用 createContext 里的 defaultValue

  • 或者通过 Context.Consumer 里只用 value 回调

# 创建 context

const Mycontext = React.createContext(defaultValue);

# context 的使用

const UserContext = React.createContext({
  nickName: "默认",
  level: -1,
});

class ProfileHeader extends Component {
  render() {
    return (
      <div>
        <h2>用户昵称：{this.context.nickname}</h2>
        <h2>用户等级：{this.context.level}</h2>
      </div>
    );
  }
}
// 使得ProfileHeader组件能够通过this.context来消费最近Context上的值
ProfileHeader.contextType = UserContext;

class Profile extends Component {
  render() {
    return (
      <div>
        <ProfileHeader />
      </div>
    );
  }
}

export default class App extends Component {
  render() {
    return (
      <div>
        <UserContext.Provider value={{ nickName: "thornsh", level: 99 }}>
          <Profile />
        </UserContext.Provider>
      </div>
    );
  }
}

# 嵌套使用多个 context

const ThemeContext = React.createContext({color:"black"});

// Provider嵌套
<UserContext.Provider value={{nickName: "thornsh", level: 99}}>
	<ThemeContext.Provider value={{color: "red"}}>
    	<Profile />
    </ThemeContext.Provider>
</UserContext.Provider>

// Consumer嵌套
<UserContext.Consumer>
	{
        value => {
            return (
            	<ThemeContext.Consumer>
                	{
                        theme => (
                        	<div>
                            	<h2 style={theme}>用户昵称：{value.nickName}</h2>
                                <h2 style={theme}>用户等级：{value.level}</h2>
                            </div>
                        )
                    }
                </ThemeContext.Consumer>
            )
        }
    }
</UserContext.Consumer>
